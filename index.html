Quick start

Python
JavaScript
cURL
import openai

openai.api_base = "https://arxon.armanxai.com/api/v1"
openai.api_key  = "sk-your-key-here"

response = openai.ChatCompletion.create(
    model="armanxai/main",
    messages=[{"role": "user", "content": "Hello, Arxon!"}],
    stream=True
)

for chunk in response:
    print(chunk.choices[0].delta.get("content", ""), end="")
